FUNCTIONStored Function — get_severity_count()
-- STORED FUNCTION: Count accidents by severity
DELIMITER $$
CREATE FUNCTION get_severity_count(sev VARCHAR(20))
RETURNS INT DETERMINISTIC READS SQL DATA
BEGIN
    DECLARE cnt INT DEFAULT 0;
    SELECT COUNT(*) INTO cnt FROM accidents WHERE severity = sev;
    RETURN cnt;
END $$

CURSORCursor — Hospital bed classification (DECLARE → OPEN → FETCH → CLOSE)
-- CURSOR: Full lifecycle — DECLARE, OPEN, FETCH, CLOSE
DELIMITER $$
CREATE PROCEDURE classify_hospital_beds()
BEGIN
    DECLARE v_id INT;
    DECLARE v_name VARCHAR(100);
    DECLARE v_beds INT;
    DECLARE done INT DEFAULT FALSE;

    -- Step 1: DECLARE the cursor
    DECLARE hospital_cursor CURSOR FOR
        SELECT hospital_id, name, available_beds FROM hospitals;

    -- Handler to detect end of result set
    DECLARE CONTINUE HANDLER FOR NOT FOUND SET done = TRUE;

    -- Temp table to store results
    CREATE TEMPORARY TABLE IF NOT EXISTS bed_report (
        hospital_id INT, name VARCHAR(100),
        available_beds INT, bed_status VARCHAR(20)
    );

    -- Step 2: OPEN the cursor
    OPEN hospital_cursor;

    -- Step 3: FETCH rows one by one in a loop
    read_loop: LOOP
        FETCH hospital_cursor INTO v_id, v_name, v_beds;
        IF done THEN LEAVE read_loop; END IF;

        INSERT INTO bed_report VALUES (
            v_id, v_name, v_beds,
            CASE
                WHEN v_beds > 50 THEN 'Available'
                WHEN v_beds > 10 THEN 'Limited'
                ELSE 'Critical'
            END
        );
    END LOOP;

    -- Step 4: CLOSE the cursor
    CLOSE hospital_cursor;

    SELECT * FROM bed_report;
    DROP TEMPORARY TABLE bed_report;
END $$
DELIMITER ;

-- Execute: CALL classify_hospital_beds();

GOLDEN HOURGolden Hour Tracking — Critical time window
-- GOLDEN HOUR: Track 60-minute critical window for high-severity accidents
SELECT a.accident_id, a.location, a.severity, a.accident_time,
       TIMESTAMPDIFF(MINUTE, a.accident_time, NOW()) AS minutes_elapsed,
       GREATEST(0, 60 - TIMESTAMPDIFF(MINUTE, a.accident_time, NOW()))
           AS golden_minutes_remaining,
       CASE
           WHEN TIMESTAMPDIFF(MINUTE, a.accident_time, NOW()) <= 60
               THEN 'WITHIN GOLDEN HOUR'
           ELSE 'GOLDEN HOUR EXPIRED'
       END AS golden_status
FROM accidents a
WHERE a.severity IN ('Critical', 'High')
ORDER BY a.accident_time DESC LIMIT 5;

HOTSPOTHotspot Analysis — High-risk locations
-- HOTSPOT ANALYSIS: Identify high-risk accident locations
SELECT location, COUNT(*) AS accident_count,
       GROUP_CONCAT(DISTINCT severity ORDER BY severity) AS severity_types,
       ROUND(AVG(CASE
           WHEN severity='Critical' THEN 4  WHEN severity='High' THEN 3
           WHEN severity='Medium' THEN 2    ELSE 1
       END), 1) AS avg_severity_score
FROM accidents
GROUP BY location HAVING COUNT(*) >= 2
ORDER BY accident_count DESC, avg_severity_score DESC;
